#include "app_2048.h"
#include <stdexcept>

// Inicjalizacja stałej nazwy klasy okna [cite: 95-96]
std::wstring const app_2048::s_class_name{ L"2048 Window" };

// Rejestracja "przepisu" na okno w systemie [cite: 97-112]
bool app_2048::register_class()
{
    WNDCLASSEXW desc{};
    if (GetClassInfoExW(m_instance, s_class_name.c_str(), &desc) != 0)
        return true;

    desc = {
        .cbSize = sizeof(WNDCLASSEXW),
        .lpfnWndProc = window_proc_static,
        .hInstance = m_instance,
        .hCursor = LoadCursorW(nullptr, L"IDC_ARROW"),
        .lpszClassName = s_class_name.c_str()
    };

    return RegisterClassExW(&desc) != 0;
}

// Budowa fizycznego okna [cite: 113-129]
HWND app_2048::create_window()
{
    return CreateWindowExW(
        0, /*empty extended styles*/
        s_class_name.c_str(),
        L"2048",
        WS_OVERLAPPED | WS_SYSMENU | WS_CAPTION | WS_BORDER | WS_MINIMIZEBOX,
        CW_USEDEFAULT, 0, /*default position*/
        CW_USEDEFAULT, 0, /*default size*/
        nullptr,
        nullptr,
        m_instance,
        this // <-- Przekazanie wskaźnika obiektu do okna
    );
}

// "Recepcjonistka": Przechwycenie wskaźnika this i delegacja komunikatów [cite: 131-161]
LRESULT app_2048::window_proc_static(
    HWND window, UINT message,
    WPARAM wparam, LPARAM lparam)
{
    app_2048 *app = nullptr;
    if (message == WM_NCCREATE)
    {
        auto p = reinterpret_cast<LPCREATESTRUCTW>(lparam);
        app = static_cast<app_2048 *>(p->lpCreateParams);
        SetWindowLongPtrW(window, GWLP_USERDATA,
            reinterpret_cast<LONG_PTR>(app));
    }
    else
    {
        app = reinterpret_cast<app_2048 *>(
            GetWindowLongPtrW(window, GWLP_USERDATA));
    }

    if (app != nullptr)
    {
        return app->window_proc(window, message, wparam, lparam);
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

// "Szef": Faktyczna obsługa zdarzeń okna [cite: 162-179]
LRESULT app_2048::window_proc(
    HWND window, UINT message,
    WPARAM wparam, LPARAM lparam)
{
    switch (message) {
    case WM_CLOSE:
        DestroyWindow(window);
        return 0;
    case WM_DESTROY:
        if (window == m_main)
            PostQuitMessage(EXIT_SUCCESS);
        return 0;
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

// Konstruktor aplikacji [cite: 180-187]
app_2048::app_2048(HINSTANCE instance)
    : m_instance{ instance }, m_main{}
{
    register_class();
    m_main = create_window();
}

// Pętla komunikatów utrzymująca program przy życiu [cite: 188-202]
int app_2048::run(int show_command)
{
    ShowWindow(m_main, show_command);
    MSG msg{};
    BOOL result = TRUE;

    while ((result = GetMessageW(&msg, nullptr, 0, 0)) != 0)
    {
        if (result == -1)
            return EXIT_FAILURE;

        TranslateMessage(&msg);
        DispatchMessageW(&msg);
    }
    return EXIT_SUCCESS;
}

LRESULT CALLBACK window_proc_static(HWND window, UINT message, WPARAM wparam, LPARAM lparam) {
    app* moj_obiekt = nullptr;

    if (message == WM_NCCREATE) {
        // 1. Złap ładunek (this) przy tworzeniu okna
        auto p = reinterpret_cast<LPCREATESTRUCTW>(lparam);
        moj_obiekt = static_cast<app*>(p->lpCreateParams);
        // 2. Schowaj go głęboko w pamięci okna
        SetWindowLongPtrW(window, GWLP_USERDATA, reinterpret_cast<LONG_PTR>(moj_obiekt));
    } else {
        // 3. Przy każdym innym kliknięciu/ruchu, wyciągnij obiekt z pamięci okna
        moj_obiekt = reinterpret_cast<app*>(GetWindowLongPtrW(window, GWLP_USERDATA));
    }

    // 4. Przekaż robotę do właściwej metody wewnątrz obiektu
    if (moj_obiekt != nullptr) {
        return moj_obiekt->window_proc(window, message, wparam, lparam);
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

3. Obsługa Klawiatury i Myszki (Procedura Klasowa)

To tutaj lądują zdarzenia. Pamiętaj o dodawaniu klamerek { } dla każdego przypadku case, jeśli tworzysz w nim nowe zmienne!

LRESULT app::window_proc(HWND window, UINT message, WPARAM wparam, LPARAM lparam) {
    switch (message) {

        // --- ZAMYKANIE APLIKACJI ---
        case WM_CLOSE:
            DestroyWindow(window);
            return 0;

        case WM_DESTROY:
            PostQuitMessage(EXIT_SUCCESS);
            return 0;

        // --- MYSZKA: KLIKNIĘCIE ---
        case WM_LBUTTONDOWN:
        {
            // LOWORD wyciąga dół (X), HIWORD wyciąga górę (Y) z lparam
            int x = LOWORD(lparam);
            int y = HIWORD(lparam);

            std::wstring tekst = L"Klik: X=" + std::to_wstring(x) + L", Y=" + std::to_wstring(y);
            SetWindowTextW(window, tekst.c_str());
            return 0;
        }

        // --- KLAWIATURA: KOD KLAWISZA (np. strzałki, WASD) ---
        case WM_KEYDOWN:
        {
            // Przydatne stałe: VK_UP, VK_DOWN, VK_LEFT, VK_RIGHT, VK_ESCAPE
            if (wparam == VK_ESCAPE) {
                DestroyWindow(window); // Wyjdź z gry po wciśnięciu ESC
            }
            return 0;
        }

        // --- KLAWIATURA: WŁAŚCIWY ZNAK (uwzględnia Shift/Alt) ---
        case WM_CHAR:
        {
            std::wstring tekst = L"Wpisano znak: ";
            tekst += static_cast<wchar_t>(wparam); // Używamy += aby bezpiecznie dokleić
            SetWindowTextW(window, tekst.c_str());
            return 0;
        }
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

4. Rysowanie (GDI - Graphics Device Interface)

UWAGA NA WYCIEKI PAMIĘCI! GDI ma ograniczoną pulę zasobów w Windowsie. Jeśli tworzysz pędzel (Brush) lub pióro (Pen), MUSISZ je potem usunąć za pomocą DeleteObject, w przeciwnym razie zawiesisz aplikację (lub cały system). 
case WM_PAINT:
{
    PAINTSTRUCT ps;
    // 1. Otwieramy płótno (Pobieramy Device Context - HDC)
    HDC dc = BeginPaint(window, &ps);

    // 2. Tworzymy pędzel (wnętrze) i pióro (ramka)
    HBRUSH mojPedzel = CreateSolidBrush(RGB(250, 192, 174));
    HPEN mojePioro = CreatePen(PS_SOLID, 2, RGB(0, 0, 0));

    // 3. Wybieramy nasze narzędzia do ręki (i zapisujemy te, które tam były)
    HBRUSH staryPedzel = (HBRUSH)SelectObject(dc, mojPedzel);
    HPEN starePioro = (HPEN)SelectObject(dc, mojePioro);

    // 4. RYSOWANIE
    // Narysuj prostokąt z zaokrąglonymi rogami
    RoundRect(dc, 50, 50, 150, 150, 11, 11);

    // 5. Sprzątanie: Odkładamy stare narzędzia na miejsce
    SelectObject(dc, staryPedzel);
    SelectObject(dc, starePioro);

    // 6. Niszczymy NASZE narzędzia, zwalniając pamięć
    DeleteObject(mojPedzel);
    DeleteObject(mojePioro);

    // 7. Zamykamy płótno
    EndPaint(window, &ps);
    return 0;
}

5. Słownik Najważniejszych Funkcji WinAPI

Funkcja Co robi? Częste Zastosowanie
RegisterClassExW Wysyła "przepis" na wygląd okna do systemu Windows. Używana tylko raz na starcie aplikacji.
CreateWindowExW Fizycznie tworzy okno z gotowego przepisu. Zwraca uchwyt okna (HWND). Inicjalizacja głównego UI.
ShowWindow Sprawia, że niewidzialne okno z pamięci RAM pojawia się na monitorze. Zaraz po utworzeniu okna głównego.
GetMessageW Usypia program do czasu, aż użytkownik coś zrobi (ruszy myszką, kliknie). Serce pętli głównej (pętli komunikatów).
SetTimer Tworzy budzik, który będzie wysyłał komunikat WM_TIMER co X milisekund. Zegary, proste animacje, odświeżanie gry.
AdjustWindowRectEx Oblicza, jak duże musi być całe okno (razem z ramką), żeby obszar roboczy w środku miał konkretne rozmiary. Przydatne do sztywnego ustalenia wymiarów planszy (np. 4x4 kafelki) przed wywołaniem CreateWindowExW.
InvalidateRect Mówi systemowi: "Ekran jest nieaktualny, wymuś natychmiastowe przemalowanie (wyślij WM_PAINT)". Wywoływane po ruchu gracza, żeby zaktualizować grafikę na ekranie.

6. Zarządzanie Oknem (Wymiary, Pozycja, Tytuł)

W trakcie działania programu (np. gry) często musisz zmienić rozmiar okna, przesunąć je na środek ekranu lub zaktualizować wynik na pasku tytułu.

Zmiana Tytułu Okna (W locie)

// Przydatne np. do pokazywania czasu gry lub wyniku na pasku.
std::wstring nowyTytul = L"2048 - Wynik: " + std::to_wstring(punkty);
SetWindowTextW(window, nowyTytul.c_str());

Zmiana Rozmiaru i Pozycji (SetWindowPos)

To potężna funkcja. Używasz specjalnych flag (zaczynających się od SWP_), aby powiedzieć systemowi, które wartości ma zignorować, a które zmienić.

// Składnia: SetWindowPos(uchwyt, Z-order, X, Y, Szerokość, Wysokość, Flagi)

// Przykład 1: Zmień TYLKO ROZMIAR na 800x600 (ignoruj podane pozycje X i Y)
SetWindowPos(window, nullptr, 0, 0, 800, 600, SWP_NOMOVE | SWP_NOZORDER);

// Przykład 2: Zmień TYLKO POZYCJĘ na X=100, Y=100 (ignoruj podany rozmiar)
SetWindowPos(window, nullptr, 100, 100, 0, 0, SWP_NOSIZE | SWP_NOZORDER);

Pobieranie Wymiarów Okna

Pamiętaj o różnicy między obszarami!
  • GetClientRect(window, &rect) - Zwraca rozmiar samego "płótna" w środku okna (obszaru klienta, po którym rysujesz). X i Y są tu zawsze równe 0.
  • GetWindowRect(window, &rect) - Zwraca pozycję i wymiary całego okna (razem z ramkami i paskiem) względem monitora.
  • DwmGetWindowAttribute(...) - Bardziej zaawansowana wersja GetWindowRect. W nowym Windowsie ramki mają niewidzialne cienie (drop shadow), które zaburzają zwykłe obliczenia. Funkcja DWM pobiera idealnie dokładne granice "widzialnej" ramki.

Idealne dopasowanie okna do planszy gry (AdjustWindowRectEx)

Jeśli wiesz, że Twoja plansza do gry ma wymiary 400x400 pikseli, nie możesz wywołać CreateWindowExW z rozmiarem 400x400, bo wtedy grube ramki i pasek tytułu "zjedzą" część Twojej planszy. Ta funkcja rozwiązuje ten problem.

// 1. Zapisujemy pożądany rozmiar SAMEJ PLANSZY (obszaru klienta)
RECT rozmiar = { 0, 0, 400, 400 };

// 2. Zapisujemy styl okna (musi być taki sam jak w CreateWindowExW!)
DWORD styl = WS_OVERLAPPED | WS_SYSMENU | WS_CAPTION | WS_MINIMIZEBOX;

// 3. Magia: System przelicza i POWIĘKSZA 'rozmiar' o grubość ramek i paska
AdjustWindowRectEx(&rozmiar, styl, FALSE, 0);

// 4. Wyciągamy rzeczywistą szerokość i wysokość całego okna do CreateWindowExW
int gotowaSzerokosc = rozmiar.right - rozmiar.left;
int gotowaWysokosc = rozmiar.bottom - rozmiar.top;

8. Obsługa Menu (WM_COMMAND)

Jeśli dodasz do okna klasyczne menu u góry (Plik, Edycja, Nowa Gra), system będzie wysyłał komunikat WM_COMMAND za każdym razem, gdy użytkownik coś w nim kliknie[cite: 657, 699].

case WM_COMMAND:
{
    // LOWORD z wparam zawiera ID klikniętego elementu menu (np. ID_NEWGAME z pliku resource.h)
    int menuId = LOWORD(wparam);

    switch (menuId) {
        case ID_NEWGAME: // Załóżmy, że zdefiniowałeś to w zasobach programu [cite: 670, 700-701]
            // Kod resetujący planszę 2048
            break;
        case ID_EXIT:
            DestroyWindow(window);
            break;
    }
    return 0;
}

10. Rysowanie Tekstu i Przezroczystość

Oprócz malowania kwadratów (GDI), musisz na nich rysować cyfry (2, 4, 8, 2048)[cite: 851, 903]. Dodatkowo okno "popup" lub ekrany wygranej mogą być półprzezroczyste[cite: 488, 885].

Przezroczyste Okno (Layered Window)

// 1. Przy tworzeniu okna (CreateWindowExW) musisz dodać na początku styl WS_EX_LAYERED [cite: 487, 501]
// 2. Następnie wywołujesz tę funkcję, by ustawić poziomzezroczystości (0 = niewidzialne, 255 = pełne)
SetLayeredWindowAttributes(window, 0, 128, LWA_ALPHA); // 128 to 50% przezroczystości [cite: 502, 524-525]

Rysowanie wyśrodkowanego tekstu (DrawText) w WM_PAINT

// Ten kod umieszczamy wewnątrz WM_PAINT, mając już HDC dc [cite: 768, 772]

std::wstring napis = L"2048";
RECT poleKafelka = { 10, 10, 70, 70 }; // Gdzie ma być narysowany tekst

// Ustawiamy tło tekstu na przezroczyste, żeby nie nadpisywało koloru naszego klocka
SetBkMode(dc, TRANSPARENT);
// Ustawiamy kolor czcionki na czarny
SetTextColor(dc, RGB(0, 0, 0));

// Rysujemy tekst wyśrodkowany w pionie i poziomie [cite: 903]
DrawTextW(dc, napis.c_str(), -1, &poleKafelka, DT_CENTER | DT_VCENTER | DT_SINGLELINE);

16. Ekstra: Dynamiczne tworzenie i poruszanie oknem-dzieckiem

Świetny mechanizm do tworzenia własnych "pionków", przycisków czy prostych gier. Zamiast rysować po płótnie (GDI), powołujemy do życia systemową kontrolkę (np. STATIC), którą możemy dowolnie skalować i przesuwać za pomocą klawiatury.

Krok 1: Zmienne w pliku nagłówkowym (.h)

private:
    // Pamięć do rysowania myszką
    POINT m_klikStart{};

    // Pamięć o naszym dynamicznym obiekcie
    int m_obiektX = 0;
    int m_obiektY = 0;
    HWND m_uchwytDziecka = nullptr; // Na starcie puste

Krok 2: Rysowanie myszką (Powołanie obiektu do życia)

Wciskając lewy przycisk, zapisujemy punkt startowy. Puszczając, liczymy szerokość i wysokość. Następnie tworzymy okno-dziecko, a jeśli już istnieje – tylko aktualizujemy jego rozmiar.

    case WM_LBUTTONDOWN:
    {
        m_klikStart.x = LOWORD(lparam);
        m_klikStart.y = HIWORD(lparam);
        return 0;
    }

    case WM_LBUTTONUP:
    {
        int koniecX = LOWORD(lparam);
        int koniecY = HIWORD(lparam);

        // Zabezpieczenie: szukamy lewego górnego rogu (zawsze mniejsza wartość)
        m_obiektX = std::min(m_klikStart.x, koniecX);
        m_obiektY = std::min(m_klikStart.y, koniecY);

        // Różnica bezwzględna daje nam wymiary
        int szerokosc = std::abs(koniecX - m_klikStart.x);
        int wysokosc = std::abs(koniecY - m_klikStart.y);

        // Tworzymy kontrolkę (jeśli to pierwszy raz)
        if (m_uchwytDziecka == nullptr) {
            m_uchwytDziecka = CreateWindowExW(
                0, L"STATIC", nullptr,
                WS_CHILD | WS_VISIBLE,
                m_obiektX, m_obiektY, szerokosc, wysokosc,
                window, // Okno rodzic
                nullptr, m_instance, nullptr
            );
        }
        // Lub zmieniamy jej wymiary (jeśli już istnieje)
        else {
            SetWindowPos(m_uchwytDziecka, nullptr,
                m_obiektX, m_obiektY, szerokosc, wysokosc,
                SWP_NOZORDER);
        }

        return 0;
    }

Krok 3: Poruszanie obiektem z klawiatury

Używamy funkcji SetWindowPos z flagą SWP_NOSIZE, która każe systemowi zignorować szerokość i wysokość, a zmienić tylko pozycję X i Y.

    case WM_KEYDOWN:
    {
        // Blokujemy ruch, jeśli użytkownik jeszcze nie narysował obiektu
        if (m_uchwytDziecka != nullptr) {
            int predkosc = 10;

            switch (wparam) {
                case 'W': m_obiektY -= predkosc; break;
                case 'S': m_obiektY += predkosc; break;
                case 'A': m_obiektX -= predkosc; break;
                case 'D': m_obiektX += predkosc; break;
            }

            // Przesuwamy okno-dziecko w nowe miejsce (ignorujemy zera, bo jest flaga NOSIZE)
            SetWindowPos(m_uchwytDziecka, nullptr,
                m_obiektX, m_obiektY, 0, 0,
                SWP_NOSIZE | SWP_NOZORDER);
        }
        return 0;
    }

Krok 4: Kolorowanie kontrolki (Opcjonalne)

Aby pozbyć się nudnego, szarego koloru systemu, nasłuchujemy zapytania o kolor.

    case WM_CTLCOLORSTATIC:
    {
        // Upewniamy się, że system pyta o NASZ konkretny obiekt
        if (reinterpret_cast<HWND>(lparam) == m_uchwytDziecka) {
            // W profesjonalnym kodzie pędzel tworzymy raz w konstruktorze
            static HBRUSH kolorPedzla = CreateSolidBrush(RGB(100, 200, 255));
            return reinterpret_cast<INT_PTR>(kolorPedzla);
        }
        break;
    }

17. Zmiana Koloru Tła Okna

Tło głównego okna w WinAPI jest definiowane przez pędzel (Brush) przypisany do całej klasy okna (tzw. przepisu). Możemy ten pędzel ustawić raz na początku, albo podmieniać go w trakcie działania programu.

Sposób 1: Na stałe (Statycznie podczas rejestracji)

Najlepsza metoda, jeśli okno ma mieć jeden konkretny kolor przez całe swoje życie. Dodajemy pędzel do struktury WNDCLASSEXW w funkcji rejestrującej okno.

// Wypełnianie struktury (np. w funkcji register_class)
desc = {
    .cbSize = sizeof(WNDCLASSEXW),
    .lpfnWndProc = window_proc_static,
    .hInstance = m_instance,
    // ... inne pola ...

    // Tworzymy i przypisujemy nowy pędzel (np. kremowy kolor gry 2048)
    .hbrBackground = CreateSolidBrush(RGB(250, 248, 239)),

    .lpszClassName = s_class_name.c_str()
};

Sposób 2: W locie (Dynamicznie np. po kliknięciu myszą)

Jeśli chcemy zmienić kolor tła w trakcie działania programu (np. po kliknięciu prawym przyciskiem myszy), musimy podmienić pędzel w pamięci klasy okna za pomocą SetClassLongPtrW.

PAMIĘTAJ O SPRZĄTANIU! Zanim wstawisz nowy pędzel do klasy okna, musisz wyciągnąć stary i go zniszczyć (DeleteObject). Inaczej przy każdym kliknięciu będziesz pożerać pamięć RAM (wyciek pamięci)! 
    case WM_RBUTTONDOWN:
    {
        // 1. Tworzymy nowy pędzel z nowym kolorem (np. jasny zielony)
        HBRUSH nowyPedzel = CreateSolidBrush(RGB(150, 255, 150));

        // 2. Wyciągamy stary pędzel z "szufladki" tła (GCLP_HBRBACKGROUND)
        HBRUSH staryPedzel = reinterpret_cast<HBRUSH>(GetClassLongPtrW(window, GCLP_HBRBACKGROUND));

        // 3. Niszczymy stary pędzel, zwalniając pamięć
        if (staryPedzel != nullptr) {
            DeleteObject(staryPedzel);
        }

        // 4. Wrzucamy nasz nowy pędzel na miejsce starego
        SetClassLongPtrW(window, GCLP_HBRBACKGROUND, reinterpret_cast<LONG_PTR>(nowyPedzel));

        // 5. Wymuszamy przerysowanie okna (TRUE oznacza "zmaż stare tło nowym pędzlem")
        InvalidateRect(window, nullptr, TRUE);

        return 0;
    }

19. Płynny ruch obiektu (Podejście GDI - Rectangle)

To jest docelowy i najbardziej profesjonalny sposób tworzenia gier w WinAPI. Zamiast używać ciężkich okien-dzieci, traktujemy nasze główne okno jak czystą kartkę i w każdej klatce rysujemy obiekt w nowym miejscu.

Uwaga na odwróconą oś Y! W matematyce punkt (0,0) jest na dole. W WinAPI (i grafice komputerowej) punkt (0,0) znajduje się w lewym górnym rogu ekranu. Oś X rośnie w prawo, a oś Y rośnie w dół. Dlatego ruch w górę oznacza zmniejszanie wartości Y.

Krok 1: Pamięć o stanie gracza (Plik nagłówkowy .h)

private:
    // Pozycja i rozmiar naszego "pionka"
    int m_graczX = 100;
    int m_graczY = 100;
    int m_rozmiar = 50;

Krok 2: Sterowanie i wymuszenie nowej klatki (WM_KEYDOWN)

Gdy użytkownik wciska klawisz, my tylko zmieniamy liczby w pamięci, a potem rozkazujemy systemowi przerysować okno (InvalidateRect).

    case WM_KEYDOWN:
    {
        int predkosc = 10; // O ile pikseli przesuwamy się co wciśnięcie

        switch (wparam) {
            case 'W': m_graczY -= predkosc; break; // W górę (Y maleje)
            case 'S': m_graczY += predkosc; break; // W dół (Y rośnie)
            case 'A': m_graczX -= predkosc; break; // W lewo (X maleje)
            case 'D': m_graczX += predkosc; break; // W prawo (X rośnie)
        }

        // TRUE oznacza: Zanim narysujesz nową klatkę, zmaż starą (użyj pędzla tła)
        InvalidateRect(window, nullptr, TRUE);
        return 0;
    }

Krok 3: Rysowanie w nowym miejscu (WM_PAINT)

Nasz malarz (Device Context) nie wie nic o klawiaturze. Po prostu bierze aktualne współrzędne X i Y z pamięci i maluje tam prostokąt.

    case WM_PAINT:
    {
        PAINTSTRUCT ps;
        HDC dc = BeginPaint(window, &ps);

        // Tworzymy i wciskamy malarzowi do ręki pędzel gracza (np. niebieski)
        HBRUSH pedzelGracza = CreateSolidBrush(RGB(100, 150, 255));
        HBRUSH staryPedzel = (HBRUSH)SelectObject(dc, pedzelGracza);

        // Rysujemy gracza! (LewyX, GórnyY, PrawyX, DolnyY)
        Rectangle(dc, m_graczX, m_graczY, m_graczX + m_rozmiar, m_graczY + m_rozmiar);

        // Odkładamy pędzle na miejsce
        SelectObject(dc, staryPedzel);
        DeleteObject(pedzelGracza);
        EndPaint(window, &ps);
        return 0;
    }

22. Słownik Podstawowych Stylów Okna (Window Styles)

Trzeci argument w funkcji CreateWindowExW (lub czwarty w CreateWindow) definiuje wygląd i zachowanie okna. Style łączymy ze sobą za pomocą operatora bitowego OR (|). Oto najważniejsze flagi z dokumentacji WinAPI:

Flaga (Styl) Co dokładnie robi?
WS_OVERLAPPED Tworzy standardowe okno główne aplikacji (zazwyczaj ma pasek tytułu i ramkę). Podstawowa flaga na start.
WS_POPUP Tworzy okno typu pop-up. Nie można go łączyć ze stylem WS_CHILD. Często używane do okienek dialogowych.
WS_CHILD Tworzy okno-dziecko (np. przycisk lub pole tekstowe). Musi mieć przypisane okno-rodzica. Nie ma paska menu.
WS_VISIBLE Okno jest domyślnie widoczne zaraz po utworzeniu (nie trzeba wywoływać ShowWindow).
WS_CAPTION Dodaje oknu pasek tytułu (pozwala to na chwytanie i przesuwanie okna myszką).
WS_SYSMENU Dodaje menu systemowe w lewym górnym rogu oraz ikonę zamknięcia ("X") w prawym górnym rogu (wymaga użycia z WS_CAPTION).
WS_MINIMIZEBOX Dodaje przycisk minimalizacji do paska tytułu (wymaga WS_SYSMENU).
WS_MAXIMIZEBOX Dodaje przycisk maksymalizacji do paska tytułu (wymaga WS_SYSMENU).
WS_THICKFRAME Dodaje grubą ramkę wokół okna. To właśnie ta flaga pozwala użytkownikowi na swobodne rozciąganie okna za pomocą myszy.
WS_BORDER Dodaje cienką ramkę do okna (nie da się za nią rozciągać).
WS_CLIPCHILDREN Optymalizacja rysowania. Mówi oknu głównemu: "Kiedy się odświeżasz, omijaj miejsca, w których leżą twoje okna-dzieci". Zapobiega to migotaniu (flickering).
WS_VSCROLL / WS_HSCROLL Dodaje do okna odpowiednio pionowy i poziomy pasek przewijania.

Style złożone (Makra)

Programiści Windowsa przygotowali gotowe zbitki najczęściej używanych flag:

  • WS_OVERLAPPEDWINDOW - Równowartość połączenia: WS_OVERLAPPED | WS_CAPTION | WS_SYSMENU | WS_THICKFRAME | WS_MINIMIZEBOX | WS_MAXIMIZEBOX. Tworzy klasyczne, w pełni funkcjonalne okno, które można dowolnie rozciągać.
  • WS_POPUPWINDOW - Równowartość połączenia: WS_POPUP | WS_BORDER | WS_SYSMENU.
Jak stworzyć okno o stałym rozmiarze?
Aby stworzyć okno, którego nie da się rozciągnąć, po prostu złóż własny zestaw flag oparty na WS_OVERLAPPEDWINDOW, ale pomiń w nim WS_THICKFRAME oraz WS_MAXIMIZEBOX. System zablokuje wtedy możliwość zmiany wymiarów.
