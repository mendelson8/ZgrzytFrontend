#include "app_2048.h"
#include <stdexcept>

// Inicjalizacja stałej nazwy klasy okna [cite: 95-96]
std::wstring const app_2048::s_class_name{ L"2048 Window" };

// Rejestracja "przepisu" na okno w systemie [cite: 97-112]
bool app_2048::register_class()
{
    WNDCLASSEXW desc{};
    if (GetClassInfoExW(m_instance, s_class_name.c_str(), &desc) != 0)
        return true;

    desc = {
        .cbSize = sizeof(WNDCLASSEXW),
        .lpfnWndProc = window_proc_static,
        .hInstance = m_instance,
        .hCursor = LoadCursorW(nullptr, L"IDC_ARROW"),
        .lpszClassName = s_class_name.c_str()
    };

    return RegisterClassExW(&desc) != 0;
}

// Budowa fizycznego okna [cite: 113-129]
HWND app_2048::create_window()
{
    return CreateWindowExW(
        0, /*empty extended styles*/
        s_class_name.c_str(),
        L"2048",
        WS_OVERLAPPED | WS_SYSMENU | WS_CAPTION | WS_BORDER | WS_MINIMIZEBOX,
        CW_USEDEFAULT, 0, /*default position*/
        CW_USEDEFAULT, 0, /*default size*/
        nullptr,
        nullptr,
        m_instance,
        this // <-- Przekazanie wskaźnika obiektu do okna
    );
}

// "Recepcjonistka": Przechwycenie wskaźnika this i delegacja komunikatów [cite: 131-161]
LRESULT app_2048::window_proc_static(
    HWND window, UINT message,
    WPARAM wparam, LPARAM lparam)
{
    app_2048 *app = nullptr;
    if (message == WM_NCCREATE)
    {
        auto p = reinterpret_cast<LPCREATESTRUCTW>(lparam);
        app = static_cast<app_2048 *>(p->lpCreateParams);
        SetWindowLongPtrW(window, GWLP_USERDATA,
            reinterpret_cast<LONG_PTR>(app));
    }
    else
    {
        app = reinterpret_cast<app_2048 *>(
            GetWindowLongPtrW(window, GWLP_USERDATA));
    }

    if (app != nullptr)
    {
        return app->window_proc(window, message, wparam, lparam);
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

// "Szef": Faktyczna obsługa zdarzeń okna [cite: 162-179]
LRESULT app_2048::window_proc(
    HWND window, UINT message,
    WPARAM wparam, LPARAM lparam)
{
    switch (message) {
    case WM_CLOSE:
        DestroyWindow(window);
        return 0;
    case WM_DESTROY:
        if (window == m_main)
            PostQuitMessage(EXIT_SUCCESS);
        return 0;
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

// Konstruktor aplikacji [cite: 180-187]
app_2048::app_2048(HINSTANCE instance)
    : m_instance{ instance }, m_main{}
{
    register_class();
    m_main = create_window();
}

// Pętla komunikatów utrzymująca program przy życiu [cite: 188-202]
int app_2048::run(int show_command)
{
    ShowWindow(m_main, show_command);
    MSG msg{};
    BOOL result = TRUE;

    while ((result = GetMessageW(&msg, nullptr, 0, 0)) != 0)
    {
        if (result == -1)
            return EXIT_FAILURE;

        TranslateMessage(&msg);
        DispatchMessageW(&msg);
    }
    return EXIT_SUCCESS;
}

LRESULT CALLBACK window_proc_static(HWND window, UINT message, WPARAM wparam, LPARAM lparam) {
    app* moj_obiekt = nullptr;

    if (message == WM_NCCREATE) {
        // 1. Złap ładunek (this) przy tworzeniu okna
        auto p = reinterpret_cast<LPCREATESTRUCTW>(lparam);
        moj_obiekt = static_cast<app*>(p->lpCreateParams);
        // 2. Schowaj go głęboko w pamięci okna
        SetWindowLongPtrW(window, GWLP_USERDATA, reinterpret_cast<LONG_PTR>(moj_obiekt));
    } else {
        // 3. Przy każdym innym kliknięciu/ruchu, wyciągnij obiekt z pamięci okna
        moj_obiekt = reinterpret_cast<app*>(GetWindowLongPtrW(window, GWLP_USERDATA));
    }

    // 4. Przekaż robotę do właściwej metody wewnątrz obiektu
    if (moj_obiekt != nullptr) {
        return moj_obiekt->window_proc(window, message, wparam, lparam);
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

3. Obsługa Klawiatury i Myszki (Procedura Klasowa)

To tutaj lądują zdarzenia. Pamiętaj o dodawaniu klamerek { } dla każdego przypadku case, jeśli tworzysz w nim nowe zmienne!

LRESULT app::window_proc(HWND window, UINT message, WPARAM wparam, LPARAM lparam) {
    switch (message) {

        // --- ZAMYKANIE APLIKACJI ---
        case WM_CLOSE:
            DestroyWindow(window);
            return 0;

        case WM_DESTROY:
            PostQuitMessage(EXIT_SUCCESS);
            return 0;

        // --- MYSZKA: KLIKNIĘCIE ---
        case WM_LBUTTONDOWN:
        {
            // LOWORD wyciąga dół (X), HIWORD wyciąga górę (Y) z lparam
            int x = LOWORD(lparam);
            int y = HIWORD(lparam);

            std::wstring tekst = L"Klik: X=" + std::to_wstring(x) + L", Y=" + std::to_wstring(y);
            SetWindowTextW(window, tekst.c_str());
            return 0;
        }

        // --- KLAWIATURA: KOD KLAWISZA (np. strzałki, WASD) ---
        case WM_KEYDOWN:
        {
            // Przydatne stałe: VK_UP, VK_DOWN, VK_LEFT, VK_RIGHT, VK_ESCAPE
            if (wparam == VK_ESCAPE) {
                DestroyWindow(window); // Wyjdź z gry po wciśnięciu ESC
            }
            return 0;
        }

        // --- KLAWIATURA: WŁAŚCIWY ZNAK (uwzględnia Shift/Alt) ---
        case WM_CHAR:
        {
            std::wstring tekst = L"Wpisano znak: ";
            tekst += static_cast<wchar_t>(wparam); // Używamy += aby bezpiecznie dokleić
            SetWindowTextW(window, tekst.c_str());
            return 0;
        }
    }
    return DefWindowProcW(window, message, wparam, lparam);
}

4. Rysowanie (GDI - Graphics Device Interface)

UWAGA NA WYCIEKI PAMIĘCI! GDI ma ograniczoną pulę zasobów w Windowsie. Jeśli tworzysz pędzel (Brush) lub pióro (Pen), MUSISZ je potem usunąć za pomocą DeleteObject, w przeciwnym razie zawiesisz aplikację (lub cały system). 
case WM_PAINT:
{
    PAINTSTRUCT ps;
    // 1. Otwieramy płótno (Pobieramy Device Context - HDC)
    HDC dc = BeginPaint(window, &ps);

    // 2. Tworzymy pędzel (wnętrze) i pióro (ramka)
    HBRUSH mojPedzel = CreateSolidBrush(RGB(250, 192, 174));
    HPEN mojePioro = CreatePen(PS_SOLID, 2, RGB(0, 0, 0));

    // 3. Wybieramy nasze narzędzia do ręki (i zapisujemy te, które tam były)
    HBRUSH staryPedzel = (HBRUSH)SelectObject(dc, mojPedzel);
    HPEN starePioro = (HPEN)SelectObject(dc, mojePioro);

    // 4. RYSOWANIE
    // Narysuj prostokąt z zaokrąglonymi rogami
    RoundRect(dc, 50, 50, 150, 150, 11, 11);

    // 5. Sprzątanie: Odkładamy stare narzędzia na miejsce
    SelectObject(dc, staryPedzel);
    SelectObject(dc, starePioro);

    // 6. Niszczymy NASZE narzędzia, zwalniając pamięć
    DeleteObject(mojPedzel);
    DeleteObject(mojePioro);

    // 7. Zamykamy płótno
    EndPaint(window, &ps);
    return 0;
}

5. Słownik Najważniejszych Funkcji WinAPI

Funkcja Co robi? Częste Zastosowanie
RegisterClassExW Wysyła "przepis" na wygląd okna do systemu Windows. Używana tylko raz na starcie aplikacji.
CreateWindowExW Fizycznie tworzy okno z gotowego przepisu. Zwraca uchwyt okna (HWND). Inicjalizacja głównego UI.
ShowWindow Sprawia, że niewidzialne okno z pamięci RAM pojawia się na monitorze. Zaraz po utworzeniu okna głównego.
GetMessageW Usypia program do czasu, aż użytkownik coś zrobi (ruszy myszką, kliknie). Serce pętli głównej (pętli komunikatów).
SetTimer Tworzy budzik, który będzie wysyłał komunikat WM_TIMER co X milisekund. Zegary, proste animacje, odświeżanie gry.
AdjustWindowRectEx Oblicza, jak duże musi być całe okno (razem z ramką), żeby obszar roboczy w środku miał konkretne rozmiary. Przydatne do sztywnego ustalenia wymiarów planszy (np. 4x4 kafelki) przed wywołaniem CreateWindowExW.
InvalidateRect Mówi systemowi: "Ekran jest nieaktualny, wymuś natychmiastowe przemalowanie (wyślij WM_PAINT)". Wywoływane po ruchu gracza, żeby zaktualizować grafikę na ekranie.

To potężna funkcja. Używasz specjalnych flag (zaczynających się od SWP_), aby powiedzieć systemowi, które wartości ma zignorować, a które zmienić.

Pobieranie Wymiarów Okna

Pamiętaj o różnicy między obszarami!
  • GetClientRect(window, &rect) - Zwraca rozmiar samego "płótna" w środku okna (obszaru klienta, po którym rysujesz). X i Y są tu zawsze równe 0.
  • GetWindowRect(window, &rect) - Zwraca pozycję i wymiary całego okna (razem z ramkami i paskiem) względem monitora.
  • DwmGetWindowAttribute(...) - Bardziej zaawansowana wersja GetWindowRect. W nowym Windowsie ramki mają niewidzialne cienie (drop shadow), które zaburzają zwykłe obliczenia. Funkcja DWM pobiera idealnie dokładne granice "widzialnej" ramki.

Idealne dopasowanie okna do planszy gry (AdjustWindowRectEx)

Jeśli wiesz, że Twoja plansza do gry ma wymiary 400x400 pikseli, nie możesz wywołać CreateWindowExW z rozmiarem 400x400, bo wtedy grube ramki i pasek tytułu "zjedzą" część Twojej planszy. Ta funkcja rozwiązuje ten problem.

// 1. Zapisujemy pożądany rozmiar SAMEJ PLANSZY (obszaru klienta)
RECT rozmiar = { 0, 0, 400, 400 };

// 2. Zapisujemy styl okna (musi być taki sam jak w CreateWindowExW!)
DWORD styl = WS_OVERLAPPED | WS_SYSMENU | WS_CAPTION | WS_MINIMIZEBOX;

// 3. Magia: System przelicza i POWIĘKSZA 'rozmiar' o grubość ramek i paska
AdjustWindowRectEx(&rozmiar, styl, FALSE, 0);

// 4. Wyciągamy rzeczywistą szerokość i wysokość całego okna do CreateWindowExW
int gotowaSzerokosc = rozmiar.right - rozmiar.left;
int gotowaWysokosc = rozmiar.bottom - rozmiar.top;

8. Obsługa Menu (WM_COMMAND)

case WM_COMMAND:
{
    // LOWORD z wparam zawiera ID klikniętego elementu menu (np. ID_NEWGAME z pliku resource.h)

    }
    return 0;
}

Zadanie laboratoryjne wymaga dwóch okien. Aby drugie okno (tzw. Popup) zachowywało się poprawnie i nie śmieciło na pasku zadań systemu Windows, musi mieć właściciela (owner) [cite: 211-214].

Tło głównego okna w WinAPI jest definiowane przez pędzel (Brush) przypisany do całej klasy okna (tzw. przepisu). Możemy ten pędzel ustawić raz na początku, albo podmieniać go w trakcie działania programu.

Najlepsza metoda, jeśli okno ma mieć jeden konkretny kolor przez całe swoje życie. Dodajemy pędzel do struktury WNDCLASSEXW w funkcji rejestrującej okno.

Sposób 2: W locie (Dynamicznie np. po kliknięciu myszą)

PAMIĘTAJ O SPRZĄTANIU! Zanim wstawisz nowy pędzel do klasy okna, musisz wyciągnąć stary i go zniszczyć (DeleteObject). Inaczej przy każdym kliknięciu będziesz pożerać pamięć RAM (wyciek pamięci)! 
    case WM_RBUTTONDOWN:
    {
        // 1. Tworzymy nowy pędzel z nowym kolorem (np. jasny zielony)
        HBRUSH nowyPedzel = CreateSolidBrush(RGB(150, 255, 150));

        // 2. Wyciągamy stary pędzel z "szufladki" tła (GCLP_HBRBACKGROUND)
        HBRUSH staryPedzel = reinterpret_cast<HBRUSH>(GetClassLongPtrW(window, GCLP_HBRBACKGROUND));

        // 3. Niszczymy stary pędzel, zwalniając pamięć

            DeleteObject(staryPedzel);
        }

        // 4. Wrzucamy nasz nowy pędzel na miejsce starego
        SetClassLongPtrW(window, GCLP_HBRBACKGROUND, reinterpret_cast<LONG_PTR>(nowyPedzel));

        // 5. Wymuszamy przerysowanie okna (TRUE oznacza "zmaż stare tło nowym pędzlem")

        return 0;
    }

19. Płynny ruch obiektu (Podejście GDI - Rectangle)

To jest docelowy i najbardziej profesjonalny sposób tworzenia gier w WinAPI. Zamiast używać ciężkich okien-dzieci, traktujemy nasze główne okno jak czystą kartkę i w każdej klatce rysujemy obiekt w nowym miejscu.

Uwaga na odwróconą oś Y! W matematyce punkt (0,0) jest na dole. W WinAPI (i grafice komputerowej) punkt (0,0) znajduje się w lewym górnym rogu ekranu. Oś X rośnie w prawo, a oś Y rośnie w dół. Dlatego ruch w górę oznacza zmniejszanie wartości Y.

Krok 1: Pamięć o stanie gracza (Plik nagłówkowy .h)

    int m_graczX = 100;
    int m_graczY = 100;
    int m_rozmiar = 50;

Krok 2: Sterowanie i wymuszenie nowej klatki (WM_KEYDOWN)

Gdy użytkownik wciska klawisz, my tylko zmieniamy liczby w pamięci, a potem rozkazujemy systemowi przerysować okno (InvalidateRect).

    {

        switch (wparam) {
            case 'W': m_graczY -= predkosc; break; // W górę (Y maleje)
            case 'S': m_graczY += predkosc; break; // W dół (Y rośnie)
            case 'A': m_graczX -= predkosc; break; // W lewo (X maleje)
            case 'D': m_graczX += predkosc; break; // W prawo (X rośnie)
        }

        // TRUE oznacza: Zanim narysujesz nową klatkę, zmaż starą (użyj pędzla tła)
        InvalidateRect(window, nullptr, TRUE);
        return 0;
    }

Krok 3: Rysowanie w nowym miejscu (WM_PAINT)

Nasz malarz (Device Context) nie wie nic o klawiaturze. Po prostu bierze aktualne współrzędne X i Y z pamięci i maluje tam prostokąt.

    case WM_PAINT:
    {
        PAINTSTRUCT ps;
        HDC dc = BeginPaint(window, &ps);

        // Tworzymy i wciskamy malarzowi do ręki pędzel gracza (np. niebieski)
        HBRUSH pedzelGracza = CreateSolidBrush(RGB(100, 150, 255));
        HBRUSH staryPedzel = (HBRUSH)SelectObject(dc, pedzelGracza);

        // Rysujemy gracza! (LewyX, GórnyY, PrawyX, DolnyY)
        Rectangle(dc, m_graczX, m_graczY, m_graczX + m_rozmiar, m_graczY + m_rozmiar);

        // Odkładamy pędzle na miejsce
        SelectObject(dc, staryPedzel);
        DeleteObject(pedzelGracza);
        EndPaint(window, &ps);
        return 0;
    }

22. Słownik Podstawowych Stylów Okna (Window Styles)

Trzeci argument w funkcji CreateWindowExW (lub czwarty w CreateWindow) definiuje wygląd i zachowanie okna. Style łączymy ze sobą za pomocą operatora bitowego OR (|). Oto najważniejsze flagi z dokumentacji WinAPI:

Flaga (Styl) Co dokładnie robi?
WS_OVERLAPPED Tworzy standardowe okno główne aplikacji (zazwyczaj ma pasek tytułu i ramkę). Podstawowa flaga na start.
WS_POPUP Tworzy okno typu pop-up. Nie można go łączyć ze stylem WS_CHILD. Często używane do okienek dialogowych.
WS_CHILD Tworzy okno-dziecko (np. przycisk lub pole tekstowe). Musi mieć przypisane okno-rodzica. Nie ma paska menu.
WS_VISIBLE Okno jest domyślnie widoczne zaraz po utworzeniu (nie trzeba wywoływać ShowWindow).
WS_CAPTION Dodaje oknu pasek tytułu (pozwala to na chwytanie i przesuwanie okna myszką).
WS_SYSMENU Dodaje menu systemowe w lewym górnym rogu oraz ikonę zamknięcia ("X") w prawym górnym rogu (wymaga użycia z WS_CAPTION).
WS_MINIMIZEBOX Dodaje przycisk minimalizacji do paska tytułu (wymaga WS_SYSMENU).
WS_MAXIMIZEBOX Dodaje przycisk maksymalizacji do paska tytułu (wymaga WS_SYSMENU).
WS_THICKFRAME Dodaje grubą ramkę wokół okna. To właśnie ta flaga pozwala użytkownikowi na swobodne rozciąganie okna za pomocą myszy.
WS_BORDER Dodaje cienką ramkę do okna (nie da się za nią rozciągać).
WS_CLIPCHILDREN Optymalizacja rysowania. Mówi oknu głównemu: "Kiedy się odświeżasz, omijaj miejsca, w których leżą twoje okna-dzieci". Zapobiega to migotaniu (flickering).
WS_VSCROLL / WS_HSCROLL Dodaje do okna odpowiednio pionowy i poziomy pasek przewijania.

Style złożone (Makra)

  • WS_OVERLAPPEDWINDOW - Równowartość połączenia: WS_OVERLAPPED | WS_CAPTION | WS_SYSMENU | WS_THICKFRAME | WS_MINIMIZEBOX | WS_MAXIMIZEBOX. Tworzy klasyczne, w pełni funkcjonalne okno, które można dowolnie rozciągać.

3. Procedura Dialogu (DialogProc)

Dialog ma swoją własną, uproszczoną procedurę. Nie używa DefWindowProc na końcu, lecz zwraca TRUE (komunikat obsłużony) lub FALSE (systemie, zajmij się tym).

INT_PTR CALLBACK PrzybornikDialogProc(HWND hwndDlg, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
    switch (uMsg)
    {
        case WM_INITDIALOG:
            // Odpala się raz na start. Ustawiamy domyślny kolor (Radio Button).
            CheckRadioButton(hwndDlg, IDC_RADIO_CZERWONY, IDC_RADIO_NIEBIESKI, IDC_RADIO_CZERWONY);
            return TRUE;

        case WM_COMMAND:
            // Logika przycisków wewnątrz dialogu
            if (LOWORD(wParam) == IDCANCEL || LOWORD(wParam) == IDOK) { // Przycisk "X" lub wyjście
                DestroyWindow(hwndDlg);  // Zbijamy okno systemowe
                g_hPrzybornik = nullptr; // Ważne: resetujemy nasz wskaźnik!
            }
            return TRUE;
    }
    return FALSE;
}

            (LPARAM)&paczka  // <-- Przekazanie wskaźnika jako LPARAM
        );

        // 4. Gdy dialog się zamknie, sprawdzamy czy użytkownik kliknął Zapisz (IDOK)
        if (wynik == IDOK) {
            // Zmieniamy rozmiar okna używając danych, które dialog odesłał w naszej paczce!
            SetWindowPos(window, nullptr, 0, 0, paczka.szerokosc, paczka.wysokosc, SWP_NOMOVE | SWP_NOZORDER);

            // Wymuszamy odświeżenie (aby np. zaktualizować tekst na środku okna)
            InvalidateRect(window, nullptr, TRUE);
        }
    }
    return 0;
}

3. Obsługa wewnątrz Dialogu (DialogProc)

To tutaj dzieje się cała inżynieria. Okno dialogowe musi odebrać paczkę przy starcie (WM_INITDIALOG), schować ją do swojej wewnętrznej "szufladki" pamięci, a potem uzupełnić ją nowymi liczbami przed zamknięciem.

Pamiętaj! Okno modalne zamykamy TYLKO i WYŁĄCZNIE funkcją EndDialog(hwndDlg, wynik);. Funkcja DestroyWindow jest tu zabroniona! 
INT_PTR CALLBACK KonfiguratorDialogProc(HWND hwndDlg, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
    switch (uMsg)
    {
        case WM_INITDIALOG:
        {
            // 1. Odbieramy adres paczki (lParam z DialogBoxParamW)
            WymiaryOkna* adresPaczki = (WymiaryOkna*)lParam;

            // 2. Zapisujemy ten wskaźnik w prywatnej pamięci tego okna dialogowego (DWLP_USER)
            // Dzięki temu będziemy mieli do niego dostęp później, jak użytkownik kliknie Zapisz.
            SetWindowLongPtrW(hwndDlg, DWLP_USER, (LONG_PTR)adresPaczki);

            // 3. Wrzucamy liczby do pół tekstowych (EDIT). FALSE oznacza, że są to liczby bez znaku (dodatnie)
            SetDlgItemInt(hwndDlg, IDC_EDIT_SZEROKOSC, adresPaczki->szerokosc, FALSE);
            SetDlgItemInt(hwndDlg, IDC_EDIT_WYSOKOSC, adresPaczki->wysokosc, FALSE);

            return TRUE;
        }

        case WM_COMMAND:
        {
            int idAkcji = LOWORD(wParam);

            // Jeśli kliknięto "Zapisz" (domyślny przycisk OK)
            if (idAkcji == IDOK) {

                // 1. Wyciągamy nasz wskaźnik z pamięci okna
                WymiaryOkna* adresPaczki = (WymiaryOkna*)GetWindowLongPtrW(hwndDlg, DWLP_USER);

                if (adresPaczki != nullptr) {
                    // 2. Pobieramy liczby z pół EDIT i zapisujemy prosto do naszej paczki!
                    // GetDlgItemInt samo zamienia tekst ze stringa na czystego inta.
                    adresPaczki->szerokosc = GetDlgItemInt(hwndDlg, IDC_EDIT_SZEROKOSC, nullptr, FALSE);
                    adresPaczki->wysokosc = GetDlgItemInt(hwndDlg, IDC_EDIT_WYSOKOSC, nullptr, FALSE);
                }

                // 3. Zamykamy dialog zwracając IDOK
                EndDialog(hwndDlg, IDOK);
                return TRUE;
            }

            // Jeśli kliknięto "Anuluj" (lub "X" w prawym górnym rogu)
            else if (idAkcji == IDCANCEL) {
                // Zamykamy dialog, nic nie zapisując do paczki
                EndDialog(hwndDlg, IDCANCEL);
                return TRUE;
            }
            break;
        }
    }
    return FALSE;
}

4. Menu Kontekstowe i Prawy Przycisk (WM_CONTEXTMENU)

Nie używamy WM_RBUTTONDOWN! WinAPI ma specjalny komunikat WM_CONTEXTMENU, który odpala się nie tylko po kliknięciu prawym przyciskiem, ale też po wciśnięciu klawisza Menu na klawiaturze. Zawsze podaje nam współrzędne ekranowe, których wymaga pływające menu.

Trick Inżynierski: TPM_RETURNCMD
Zamiast przechwytywać kliknięcia menu w globalnym WM_COMMAND, flaga TPM_RETURNCMD blokuje program, wyświetla menu i od razu zwraca ID klikniętej opcji jako wynik funkcji. To robi kod 100x czytelniejszym! 
case WM_CONTEXTMENU:
{
    HWND hKliknieteOkno = (HWND)wparam;
    HWND hLista = GetDlgItem(window, IDC_LISTA_OBECNOSCI);

    // 1. Sprawdzamy, czy menu wywołano klikając w nasz ListBox
    if (hKliknieteOkno == hLista) {

        // Współrzędne z WM_CONTEXTMENU są ekranowe (gotowe pod menu)
        int ekranX = GET_X_LPARAM(lparam);
        int ekranY = GET_Y_LPARAM(lparam);

        // 2. Odpytujemy ListBox: "Który wiersz został kliknięty?"
        // Funkcja wymaga współrzędnych relatywnych do okna ListBoxa, więc musimy je przeliczyć:
        POINT pt = { ekranX, ekranY };
        ScreenToClient(hLista, &pt);

        // Wysyłamy zapytanie o indeks wiersza z tych współrzędnych
        DWORD wynik = SendMessageW(hLista, LB_ITEMFROMPOINT, 0, MAKELPARAM(pt.x, pt.y));

        // HIWORD wynosi 0, jeśli kliknięto w poprawny element (nie w puste tło listy)
        if (HIWORD(wynik) == 0) {

            int kliknietyIndeks = LOWORD(wynik);

            // Wymuszamy podświetlenie klikniętego wiersza na niebiesko
            SendMessageW(hLista, LB_SETCURSEL, kliknietyIndeks, 0);

            // 3. Tworzymy menu w locie z kodu
            HMENU hMenu = CreatePopupMenu();
            AppendMenuW(hMenu, MF_STRING, IDM_USUN, L"Usuń Element");
            AppendMenuW(hMenu, MF_STRING, IDM_DUPLIKUJ, L"Duplikuj Element");

            // 4. Wyświetlamy menu na współrzędnych ekranowych. 
            // TPM_RETURNCMD sprawia, że funkcja czeka i zwraca nam ID klikniętej opcji!
            int wybranaOpcja = TrackPopupMenu(hMenu, TPM_RIGHTBUTTON | TPM_RETURNCMD,
                                              ekranX, ekranY, 0, window, nullptr);

            // 5. Niszczymy szkielet menu, by nie pożerał pamięci
            DestroyMenu(hMenu);

            // 6. Wykonujemy akcję zależnie od tego, co zwrócił TrackPopupMenu
            if (wybranaOpcja == IDM_USUN) {
                SendMessageW(hLista, LB_DELETESTRING, kliknietyIndeks, 0);
            }
            else if (wybranaOpcja == IDM_DUPLIKUJ) {
                // Pobieramy tekst starego elementu i wstawiamy go jako nowy
                wchar_t bufor[256];
                SendMessageW(hLista, LB_GETTEXT, kliknietyIndeks, (LPARAM)bufor);
                SendMessageW(hLista, LB_ADDSTRING, 0, (LPARAM)bufor);
            }
        }
    }
    return 0;
}

4. Silnik Animacji (WM_TIMER)

Kiedy funkcja SetTimer jest aktywna, system co określoną liczbę milisekund wysyła do okna komunikat WM_TIMER. To tutaj losujemy nowe współrzędne dla przycisku.

case WM_TIMER:
{
    // Sprawdzamy, czy to tyknięcie naszego zegara nr 1
    if (wparam == 1) {

        // 1. Pobieramy rozmiar obszaru roboczego okna głównego (żeby przycisk nie wyleciał za ekran)
        RECT obszarOkna;
        GetClientRect(window, &obszarOkna);

        // 2. Obliczamy maksymalne bezpieczne współrzędne (Odejmujemy wymiary przycisku 100x50)
        int max_X = obszarOkna.right - 100;
        int max_Y = obszarOkna.bottom - 50;

        // 3. Losujemy nową pozycję (funkcja rand() z biblioteki <cstdlib>)
        // Używamy std::max żeby uniknąć modulo przez 0, gdyby okno było ekstremalnie małe
        int nowyX = rand() % std::max(1, max_X);
        int nowyY = rand() % std::max(1, max_Y);

        // 4. Zdobywamy uchwyt do przycisku i teleportujemy go!
        // Flaga SWP_NOSIZE sprawia, że ignorujemy parametry szerokości i wysokości w tej funkcji.
        HWND hPrzycisk = GetDlgItem(window, IDC_ZLAP_MNIE);
        SetWindowPos(hPrzycisk, nullptr, nowyX, nowyY, 0, 0, SWP_NOSIZE | SWP_NOZORDER);
    }
    return 0;
}

3. Obsługa Suwaka (WM_HSCROLL)

To najtrudniejszy element. Komunikat WM_HSCROLL podaje nam w lparam uchwyt do przesuwanego suwaka, a w wparam informację o tym, CO DOKŁADNIE robi użytkownik (np. ciągnie za kciuk suwaka).

case WM_HSCROLL:
{
    // 1. Wyciągamy uchwyt do kontrolki, która wywołała ten komunikat
    HWND hScroll = (HWND)lparam;

    // Jeśli to nie suwak wywołał zdarzenie (tylko np. pasek okna), ignorujemy
    if (hScroll == nullptr) return 0;

    // 2. Odczytujemy intencję użytkownika i obecną pozycję
    int akcja = LOWORD(wparam);
    int obecnaPozycja = GetScrollPos(hScroll, SB_CTL);

    // 3. Reagujemy na różne akcje myszki na suwaku
    switch (akcja) {
        case SB_LINELEFT: obecnaPozycja -= 1; break;  // Kliknięcie w lewą strzałkę
        case SB_LINERIGHT: obecnaPozycja += 1; break; // Kliknięcie w prawą strzałkę
        case SB_PAGELEFT: obecnaPozycja -= 10; break; // Kliknięcie w pasek po lewej
        case SB_PAGERIGHT: obecnaPozycja += 10; break;// Kliknięcie w pasek po prawej
        case SB_THUMBTRACK: obecnaPozycja = HIWORD(wparam); break; // Ciągnięcie za suwak w locie!
    }

    // 4. Zabezpieczenie przed wyjściem poza zakres 0-255
    obecnaPozycja = std::max(0, std::min(255, obecnaPozycja));

    // 5. Fizycznie przesuwamy "kciuk" suwaka na nową pozycję na ekranie
    SetScrollPos(hScroll, SB_CTL, obecnaPozycja, TRUE);

    // 6. Aktualizujemy nasze zmienne w pamięci i teksty zależnie od tego, KTÓRY to suwak
    int idSuwaka = GetDlgCtrlID(hScroll);
    if (idSuwaka == IDC_SCROLL_R) {
        m_colorR = obecnaPozycja;
        SetDlgItemInt(window, IDC_TEXT_R, m_colorR, FALSE);
    }
    else if (idSuwaka == IDC_SCROLL_G) {
        m_colorG = obecnaPozycja;
        SetDlgItemInt(window, IDC_TEXT_G, m_colorG, FALSE);
    }
    else if (idSuwaka == IDC_SCROLL_B) {
        m_colorB = obecnaPozycja;
        SetDlgItemInt(window, IDC_TEXT_B, m_colorB, FALSE);
    }

    // 7. Zmuszamy okno do ponownego narysowania prostokąta z nowym kolorem!
    InvalidateRect(window, nullptr, TRUE);

    return 0;
}

}

INT_PTR CALLBACK UstawieniaDialogProc(HWND hwndDlg, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
    switch (uMsg)
    {
        // 1. WM_INITDIALOG to odpowiednik WM_CREATE. Wykonuje się RAZ na start.
        case WM_INITDIALOG:
            // Tutaj możesz np. ustawić domyślny tekst w polu edycji wewnątrz dialogu
            return TRUE;

        // 2. WM_COMMAND to kliknięcia w przyciski WEWNĄTRZ tego małego okienka
        case WM_COMMAND:
        {
            int kliknieteID = LOWORD(wParam);

            if (kliknieteID == IDC_PRZYCISK_OK)
            {
                // Użytkownik kliknął Zapisz! 
                // Tutaj zapisujesz dane, a potem ZAMYKASZ DIALOG:
                EndDialog(hwndDlg, IDC_PRZYCISK_OK);
                return TRUE;
            }
            else if (kliknieteID == IDC_PRZYCISK_ANULUJ)
            {
                // Użytkownik rozmyślił się. Zamykamy z kodem błędu.
                EndDialog(hwndDlg, IDC_PRZYCISK_ANULUJ);
                return TRUE;
            }
            break;
        }
    }

    // Jeśli nie dotykaliśmy tego komunikatu, zwracamy FALSE
    return FALSE;
}

40. ZADANIE SPECJALNE: Tarcza Telefonu cz. 1 (Generowanie)

Rozwiązanie problemu obracających się kontrolek polega na wykorzystaniu klasy STATIC bez flagi SS_NOTIFY. Dzięki temu okienka są widoczne, ale "przezroczyste" dla kliknięć myszką, co pozwala obsłużyć całą matematykę obrotu w jednym miejscu – w oknie głównym.

Krok 1: Zmienne globalne (Na samej górze pliku)

// Pamięć naszej tarczy
HWND g_cyfry[10];           // Tablica uchwytów do naszych kafelków
double g_calkowityKat = 0;  // O ile radianów tarcza jest aktualnie obrócona
double g_katStartowy = 0;   // Pomocnicza do mierzenia ruchu myszki
bool g_czyKrece = false;    // Flaga trzymania tarczy

// Stałe wymiary (łatwiej modyfikować)
const int TARCZA_X = 300;   // Środek tarczy (X)
const int TARCZA_Y = 300;   // Środek tarczy (Y)
const int TARCZA_R = 150;   // Promień tarczy

Krok 2: Produkcja kafelków (Wewnątrz WM_CREATE)

Używamy pętli i funkcji trygonometrycznych, aby rozłożyć 10 kafelków po okręgu. Ważne: dołącz #include <string> oraz #include <cmath> na górze pliku!

case WM_CREATE:
{
    for (int i = 0; i < 10; i++)
    {
        // Wyliczamy kąt dla danej cyfry (2*PI to pełne koło, dzielimy na 10)
        double kat = i * (6.283185 / 10.0);

        // Wyliczamy pozycje X i Y. Odejmujemy 15, żeby wyśrodkować kafelek (ma wymiary 30x30)
        int x = TARCZA_X + TARCZA_R * cos(kat) - 15;
        int y = TARCZA_Y + TARCZA_R * sin(kat) - 15;

        // Zamieniamy liczbę 'i' na tekst w formacie szerokim (L"0", L"1"...)
        std::wstring napis = std::to_wstring(i);

        // Tworzymy okienko! Brak SS_NOTIFY to klucz do sukcesu.
        g_cyfry[i] = CreateWindowExW(
            WS_EX_CLIENTEDGE, L"STATIC", napis.c_str(),
            WS_CHILD | WS_VISIBLE | SS_CENTER | SS_CENTERIMAGE,
            x, y, 30, 30,
            hwnd, (HMENU)(2000 + i),
            (HINSTANCE)GetWindowLongPtr(hwnd, GWLP_HINSTANCE), nullptr
        );
    }
    return 0;
}

41. ZADANIE SPECJALNE: Tarcza Telefonu cz. 2 (Hit Testing)

Ponieważ kafelki przepuszczają kliknięcia, główne okno musi samo zgadnąć, czy użytkownik trafił w cyfrę. Używamy do tego klasycznego Twierdzenia Pitagorasa (liczymy odległość kursora od środka każdej cyfry).

case WM_LBUTTONDOWN:
{
    int myszX = GET_X_LPARAM(lParam);
    int myszY = GET_Y_LPARAM(lParam);

    // Przeszukujemy wszystkie 10 pozycji na tarczy
    for (int i = 0; i < 10; i++)
    {
        // GDZIE aktualnie leży ta cyfra? (Bierzemy pod uwagę aktualny obrót tarczy!)
        double katAktualnyCyfry = i * (6.283185 / 10.0) + g_calkowityKat;
        int srodekX = TARCZA_X + TARCZA_R * cos(katAktualnyCyfry);
        int srodekY = TARCZA_Y + TARCZA_R * sin(katAktualnyCyfry);

        // Pitagoras: odległość myszki od środka tej cyfry
        double dystans = sqrt(pow(myszX - srodekX, 2) + pow(myszY - srodekY, 2));

        // Jeśli trafiliśmy (promień kafelka to ok. 15 pikseli)
        if (dystans <= 15.0)
        {
            g_czyKrece = true;
            SetCapture(hwnd); // Łapiemy myszkę!

            // Zapisujemy, pod jakim kątem złapaliśmy tarczę (używamy funkcji atan2)
            g_katStartowy = atan2(myszY - TARCZA_Y, myszX - TARCZA_X);
            break; // Trafiliśmy, nie musimy sprawdzać reszty
        }
    }
    return 0;
}

42. ZADANIE SPECJALNE: Tarcza Telefonu cz. 3 (Kręcenie Karuzelą)

Mając zablokowaną myszkę i znając kąt startowy, możemy w czasie rzeczywistym aktualizować g_calkowityKat i fizycznie przesuwać kafelki używając ich uchwytów za pomocą funkcji SetWindowPos.

case WM_MOUSEMOVE:
{
    if (g_czyKrece)
    {
        int myszX = GET_X_LPARAM(lParam);
        int myszY = GET_Y_LPARAM(lParam);

        // 1. Liczymy nowy kąt myszki i różnicę od ostatniego pomiaru
        double katMyszki = atan2(myszY - TARCZA_Y, myszX - TARCZA_X);
        g_calkowityKat += (katMyszki - g_katStartowy);
        g_katStartowy = katMyszki; // Aktualizujemy na następny ruch!

        // 2. Przesuwamy prawdziwe okienka!
        for (int i = 0; i < 10; i++)
        {
            // Wyliczamy nową pozycję z uwzględnieniem nowego g_calkowityKat
            double nowyKat = i * (6.283185 / 10.0) + g_calkowityKat;
            int noweX = TARCZA_X + TARCZA_R * cos(nowyKat) - 15;
            int noweY = TARCZA_Y + TARCZA_R * sin(nowyKat) - 15;

            // Przesuwamy kafelek o indeksie 'i' na wyliczoną pozycję na okręgu
            SetWindowPos(g_cyfry[i], nullptr, noweX, noweY, 0, 0, SWP_NOSIZE | SWP_NOZORDER);
        }
    }
    return 0;
}

case WM_LBUTTONUP:
{
    // Puszczamy tarczę
    if (g_czyKrece)
    {
        ReleaseCapture();
        g_czyKrece = false;
    }
    return 0;
}
Genialny Inżynierski Tip:
Zwróć uwagę, że zmienne do wymiarów tarczy (TARCZA_X, TARCZA_Y, TARCZA_R) wyciągnęliśmy jako stałe na samą górę. Prowadzący na PWSG to uwielbiają. Jeśli prowadzący podejdzie i powie: "A niech pan zrobi mniejszą tarczę i przesunie ją w lewy róg", zmieniasz tylko te trzy wartości na górze kodu, a reszta matematyki robi się sama!
