Współdzielona pamięć dla opornych

Pełny cykl życia obiektów POSIX z użyciem makra ERR.

Pełny cykl SHM Semafor Nazwany Mutex (Shared) Zmienna Warunkowa (CV) Bariera (Shared) Mutex Robust mmap (Rodzina procesów)

1. Pełny cykl życia pamięci współdzielonej

Od utworzenia (shm_open), przez mapowanie (mmap), aż po zniszczenie w systemie operacyjnym (shm_unlink).

cykl_zycia_shm.c
#include #include #include #include #include "macros.h" // ================= 1. INICJALIZACJA ================= int shm_fd = shm_open("/moj_kanal", O_CREAT | O_RDWR, 0666); if (shm_fd == -1) ERR("shm_open"); if (ftruncate(shm_fd, 4096) == -1) ERR("ftruncate"); int* data = mmap(NULL, 4096, PROT_READ | PROT_WRITE, MAP_SHARED, shm_fd, 0); if (data == MAP_FAILED) ERR("mmap"); if (close(shm_fd) == -1) ERR("close"); // ... TUTAJ PROCES KORZYSTA Z PAMIĘCI 'ptr' ... // ================= 2. SPRZĄTANIE (DESTROY) ================= if (munmap(data, 4096) == -1) ERR("munmap"); if (shm_unlink("/moj_kanal") == -1) ERR("shm_unlink");
#include <fcntl.h>
#include <sys/stat.h>
#include <sys/mman.h>
#include <unistd.h>
#include "macros.h"

// ====== 1. INICJALIZACJA ======
int shm_fd = shm_open("/moj_kanal", O_CREAT | O_RDWR, 0666);
if (shm_fd == -1) ERR("shm_open");

if (ftruncate(shm_fd, 4096) == -1) ERR("ftruncate");

void* ptr = mmap(NULL, 4096, PROT_READ | PROT_WRITE, MAP_SHARED, shm_fd, 0);
if (ptr == MAP_FAILED) ERR("mmap");

if (close(shm_fd) == -1) ERR("close");

// ====== 2. SPRZĄTANIE ======
if (munmap(ptr, 4096) == -1) ERR("munmap");

// Niszczy obiekt współdzielony całkowicie
if (shm_unlink("/moj_kanal") == -1) ERR("shm_unlink");

2. Semafor Nazwany (sem_open)

Idealny do ochrony "sekcji inicjalizacji" pamięci dzielonej. Pamiętaj o sem_unlink na końcu!

semafor.c
#include #include #include #include "macros.h" sem_t* sem = sem_open("/moj_semafor", O_CREAT, 0666, 1); if (sem == SEM_FAILED) ERR("sem_open"); if (sem_wait(sem) == -1) ERR("sem_wait"); // --- SEKCJA KRYTYCZNA --- if (sem_post(sem) == -1) ERR("sem_post"); if (sem_close(sem) == -1) ERR("sem_close"); if (sem_unlink("/moj_semafor") == -1) ERR("sem_unlink");
#include <semaphore.h>
#include "macros.h"

sem_t* sem = sem_open("/moj_semafor", O_CREAT, 0666, 1);
if (sem == SEM_FAILED) ERR("sem_open");

sem_wait(sem);
// ... KRYTYCZNE (np. inicjalizacja) ...
sem_post(sem);

// Zamknięcie lokalne
if (sem_close(sem) == -1) ERR("sem_close");

// DESTROY: Zniszczenie obiektu w całym systemie
if (sem_unlink("/moj_semafor") == -1) ERR("sem_unlink");

#include "macros.h"

5. Bariera Międzyprocesowa (Shared Barrier)

Zatrzymywanie grupy procesów. Po użyciu wywołaj destroy.

#include #include "macros.h" pthread_barrier_t* barrier = (pthread_barrier_t*) ptr; pthread_barrierattr_t attr; if (pthread_barrierattr_init(&attr) != 0) ERR("pthread_barrierattr_init"); if (pthread_barrierattr_setpshared(&attr, PTHREAD_PROCESS_SHARED) != 0) ERR("pthread_barrierattr_setpshared"); if (pthread_barrier_init(barrier, &attr, 3) != 0) ERR("pthread_barrier_init"); pthread_barrierattr_destroy(&attr); int ret = pthread_barrier_wait(barrier); if (ret != 0 && ret != PTHREAD_BARRIER_SERIAL_THREAD) ERR("pthread_barrier_wait"); if (pthread_barrier_destroy(barrier) != 0) ERR("pthread_barrier_destroy");

6. Mutex Robust (Odporny na awarie)

Zapobiega zakleszczeniu całego systemu w przypadku nagłego zabicia procesu trzymającego blokadę.

mutex_robust.c
#include #include #include "macros.h" pthread_mutexattr_t attr; if (pthread_mutexattr_init(&attr) != 0) ERR("init"); if (pthread_mutexattr_setpshared(&attr, PTHREAD_PROCESS_SHARED) != 0) ERR("shared"); if (pthread_mutexattr_setrobust(&attr, PTHREAD_MUTEX_ROBUST) != 0) ERR("robust"); if (pthread_mutex_init(mtx, &attr) != 0) ERR("init"); pthread_mutexattr_destroy(&attr); int rc = pthread_mutex_lock(mtx); if (rc == EOWNERDEAD) { // Napraw struktury w pamięci if (pthread_mutex_consistent(mtx) != 0) ERR("consistent"); } else if (rc != 0) { ERR("lock"); } pthread_mutex_unlock(mtx);
#include <pthread.h>
#include <errno.h>

pthread_mutexattr_t attr;
pthread_mutexattr_init(&attr);
pthread_mutexattr_setpshared(&attr, PTHREAD_PROCESS_SHARED);
pthread_mutexattr_setrobust(&attr, PTHREAD_MUTEX_ROBUST);
pthread_mutex_init(mtx, &attr);

int rc = pthread_mutex_lock(mtx);
if (rc == EOWNERDEAD) {
    // KROK A: Napraw dane po martwym procesie
    // KROK B: Przywróć spójność mutexa
    pthread_mutex_consistent(mtx);
} else if (rc != 0) {
    ERR("lock");
}
pthread_mutex_unlock(mtx);

7. Anonimowa pamięć współdzielona (Wiele potomków)

Omija shm_open. Używa flagi MAP_ANONYMOUS oraz deskryptora -1. Kod uwzględnia tworzenie wielu dzieci w pętli for oraz użycie funkcji msync.

mmap_wiele_dzieci.c
#include #include #include #include "macros.h" #define NUM_CHILDREN 3 // 1. Zmapowanie pamięci na tablicę (np. 3 inty) size_t mem_size = NUM_CHILDREN * sizeof(int); int* data = mmap(NULL, mem_size, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS, -1, 0); if (data == MAP_FAILED) ERR("mmap"); // 2. Pętla tworząca procesy potomne for (int i = 0; i < NUM_CHILDREN; i++) { pid_t pid = fork(); if (pid == -1) ERR("fork"); if (pid == 0) { // --- POTOMEK --- // Każdy potomek pisze w swój unikalny indeks data[i] = (i + 1) * 10; // Zsynchronizowanie pamięci (wymuszenie widoczności zmian) if (msync(data, mem_size, MS_SYNC) == -1) ERR("msync"); if (munmap(data, mem_size) == -1) ERR("munmap child"); return 0; // BARDZO WAŻNE: Potomek musi zakończyć działanie, by nie wchodzić dalej w pętlę fork! } } // --- MACIERZYSTY --- // 3. Pętla oczekująca na zakończenie WSZYSTKICH dzieci for (int i = 0; i < NUM_CHILDREN; i++) { if (wait(NULL) == -1) ERR("wait"); } // Rodzic ma teraz pewność, że wszystkie dzieci skończyły. // Można odczytać wyniki: data[0]=10, data[1]=20, data[2]=30. // 4. Sprzątanie if (munmap(ptr, mem_size) == -1) ERR("munmap parent");
#include <sys/mman.h>
#include <sys/wait.h>
#include <unistd.h>
#include "macros.h"

#define NUM_CHILDREN 3

size_t mem_size = NUM_CHILDREN * sizeof(int);
void* ptr = mmap(NULL, mem_size, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS, -1, 0);
if (ptr == MAP_FAILED) ERR("mmap");

int* data = (int*)ptr;

for (int i = 0; i < NUM_CHILDREN; i++) {
    pid_t pid = fork();
    if (pid == -1) ERR("fork");

    if (pid == 0) {
        // POTOMEK: Bezpieczny zapis w swój indeks
        data[i] = (i + 1) * 10;

        // Synchronizacja pamięci z flagą MS_SYNC
        if (msync(ptr, mem_size, MS_SYNC) == -1) ERR("msync");

        if (munmap(ptr, mem_size) == -1) ERR("munmap");
        return0;
    }
}

// MACIERZYSTY: Oczekuje na wszystkie procesy potomne
for (int i = 0; i < NUM_CHILDREN; i++) {
    if (wait(NULL) == -1) ERR("wait");
}

if (munmap(ptr, mem_size) == -1) ERR("munmap");
