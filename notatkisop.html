1. Pełny cykl życia pamięci współdzielonej

Od utworzenia (shm_open), przez mapowanie (mmap), aż po zniszczenie w systemie operacyjnym (shm_unlink).

cykl_zycia_shm.c
#include #include #include #include #include "macros.h" // ================= 1. INICJALIZACJA ================= // Tworzenie/otwarcie deskryptora pliku shm int shm_fd = shm_open("/moj_kanal", O_CREAT | O_RDWR, 0666); if (shm_fd == -1) ERR("shm_open"); // Ustalenie rozmiaru pamięci (np. 4096 bajtów) if (ftruncate(shm_fd, 4096) == -1) ERR("ftruncate"); // Mapowanie deskryptora do wirtualnej pamięci procesu (ptr) void* ptr = mmap(NULL, 4096, PROT_READ | PROT_WRITE, MAP_SHARED, shm_fd, 0); if (ptr == MAP_FAILED) ERR("mmap"); // Zamknięcie deskryptora - nie jest już potrzebny po mmap, mamy wskaźnik 'ptr' if (close(shm_fd) == -1) ERR("close"); // ... TUTAJ PROCES KORZYSTA Z PAMIĘCI 'ptr' ... // ================= 2. SPRZĄTANIE (DESTROY) ================= // Odmapowanie pamięci z wirtualnej przestrzeni TYLKO TEGO procesu if (munmap(ptr, 4096) == -1) ERR("munmap"); // Całkowite zniszczenie rury/kanału w systemie (Zazwyczaj robi to tylko program czyszczący!) // Usunięcie tego sprawi, że wywołanie ls /dev/shm nie pokaże już naszego pliku if (shm_unlink("/moj_kanal") == -1) ERR("shm_unlink");
#include <fcntl.h>
#include <sys/stat.h>
#include <sys/mman.h>
#include <unistd.h>
#include "macros.h"

// ====== 1. INICJALIZACJA ======
int shm_fd = shm_open("/moj_kanal", O_CREAT | O_RDWR, 0666);
if (shm_fd == -1) ERR("shm_open");

if (ftruncate(shm_fd, 4096) == -1) ERR("ftruncate");

void* ptr = mmap(NULL, 4096, PROT_READ | PROT_WRITE, MAP_SHARED, shm_fd, 0);
if (ptr == MAP_FAILED) ERR("mmap");

if (close(shm_fd) == -1) ERR("close");

// ====== 2. SPRZĄTANIE ======
if (munmap(ptr, 4096) == -1) ERR("munmap");

// Niszczy obiekt współdzielony całkowicie (odpowiednik _destroy dla shm)
if (shm_unlink("/moj_kanal") == -1) ERR("shm_unlink");

2. Semafor Nazwany (sem_open)

Idealny do ochrony "sekcji inicjalizacji" pamięci dzielonej. Pamiętaj o sem_unlink na końcu!

semafor.c
#include #include #include #include "macros.h" // Tworzenie semafora (nazwa MUSI zaczynać się od /) sem_t* sem = sem_open("/moj_semafor", O_CREAT, 0666, 1); if (sem == SEM_FAILED) ERR("sem_open"); // Zablokowanie (dekrementacja) if (sem_wait(sem) == -1) ERR("sem_wait"); // --- SEKCJA KRYTYCZNA --- // Odblokowanie (inkrementacja) if (sem_post(sem) == -1) ERR("sem_post"); // Odłączenie procesu od semafora (analogiczne do close) if (sem_close(sem) == -1) ERR("sem_close"); // DESTROY: Całkowite usunięcie semafora z systemu (/dev/shm/) if (sem_unlink("/moj_semafor") == -1) ERR("sem_unlink");
#include <semaphore.h>
#include "macros.h"

sem_t* sem = sem_open("/moj_semafor", O_CREAT, 0666, 1);
if (sem == SEM_FAILED) ERR("sem_open");

sem_wait(sem);
// ... KRYTYCZNE (np. inicjalizacja) ...
sem_post(sem);

// Zamknięcie lokalne
if (sem_close(sem) == -1) ERR("sem_close");

// DESTROY: Zniszczenie obiektu w całym systemie
if (sem_unlink("/moj_semafor") == -1) ERR("sem_unlink");

3. Mutex Międzyprocesowy (Shared Mutex)

Zmienna leży w mmap. Zawiera obowiązkowe niszczenie na końcu.

shared_mutex.c
#include #include "macros.h" // UWAGA: wskaźnik 'mtx' musi wskazywać na adres leżący W PAMIĘCI WSPÓŁDZIELONEJ (ptr) pthread_mutex_t* mtx = (pthread_mutex_t*) ptr; pthread_mutexattr_t attr; if (pthread_mutexattr_init(&attr) != 0) ERR("pthread_mutexattr_init"); // Współdzielenie między procesami! if (pthread_mutexattr_setpshared(&attr, PTHREAD_PROCESS_SHARED) != 0) ERR("pthread_mutexattr_setpshared"); // Inicjalizacja if (pthread_mutex_init(mtx, &attr) != 0) ERR("pthread_mutex_init"); pthread_mutexattr_destroy(&attr); // Atrybuty usuwamy od razu // Użycie if (pthread_mutex_lock(mtx) != 0) ERR("pthread_mutex_lock"); // ... działanie ... if (pthread_mutex_unlock(mtx) != 0) ERR("pthread_mutex_unlock"); // DESTROY: Niszczymy mutex TYLKO RAZ przed zamknięciem całej struktury kanału (np. ops-cleanup) if (pthread_mutex_destroy(mtx) != 0) ERR("pthread_mutex_destroy");
#include <pthread.h>
#include "macros.h"

pthread_mutex_t* mtx = (pthread_mutex_t*) ptr;

pthread_mutexattr_t attr;
if (pthread_mutexattr_init(&attr) != 0) ERR("attr_init");
if (pthread_mutexattr_setpshared(&attr, PTHREAD_PROCESS_SHARED) != 0) ERR("setpshared");
if (pthread_mutex_init(mtx, &attr) != 0) ERR("init");
pthread_mutexattr_destroy(&attr);

pthread_mutex_lock(mtx);
// ... KRYTYCZNE ...
pthread_mutex_unlock(mtx);

// DESTROY: Gdy mutex przestaje być komukolwiek potrzebny
if (pthread_mutex_destroy(mtx) != 0) ERR("mutex_destroy");

5. Bariera Międzyprocesowa (Shared Barrier)

Zatrzymywanie grupy procesów. Po użyciu wywołaj destroy.

shared_barrier.c
#include #include "macros.h" // Wskaźnik wewnątrz mmap() pthread_barrier_t* barrier = (pthread_barrier_t*) ptr; pthread_barrierattr_t attr; if (pthread_barrierattr_init(&attr) != 0) ERR("pthread_barrierattr_init"); if (pthread_barrierattr_setpshared(&attr, PTHREAD_PROCESS_SHARED) != 0) ERR("pthread_barrierattr_setpshared"); // Inicjalizacja: wymaga podania liczby procesów (np. 3) if (pthread_barrier_init(barrier, &attr, 3) != 0) ERR("pthread_barrier_init"); pthread_barrierattr_destroy(&attr); // Czekanie na resztę procesów // Zwraca PTHREAD_BARRIER_SERIAL_THREAD dla jednego z wątków, dla reszty 0 int ret = pthread_barrier_wait(barrier); if (ret != 0 && ret != PTHREAD_BARRIER_SERIAL_THREAD) ERR("pthread_barrier_wait"); // DESTROY: Po zwolnieniu i zamknięciu systemu pamięci if (pthread_barrier_destroy(barrier) != 0) ERR("pthread_barrier_destroy");
pthread_barrierattr_t attr;
if (pthread_barrierattr_init(&attr) != 0) ERR("attr_init");
if (pthread_barrierattr_setpshared(&attr, PTHREAD_PROCESS_SHARED) != 0) ERR("setpshared");
if (pthread_barrier_init(barrier, &attr, 3) != 0) ERR("barrier_init");
pthread_barrierattr_destroy(&attr);

// Blokada do momentu przyjścia 3 procesów
pthread_barrier_wait(barrier);

// DESTROY: Czyszczenie pamięci po barierze
if (pthread_barrier_destroy(barrier) != 0) ERR("barrier_destroy");
